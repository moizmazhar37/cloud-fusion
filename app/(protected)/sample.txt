jobs:
  deploy:
    runs-on: ubuntu-latest
    
    env:
      CI: false
      
    steps:
      - name: Set up Node.js
        uses: actions/setup-node@v2
        with:
          node-version: 16
      - name: Checkout code
        uses: actions/checkout@v2
      
      - name: install dependencies
        run: npm ci --legacy-peer-deps
        # run: npm cache clean --force && npm ci --legacy-peer-deps && npm run build
    
      - name: Build file
        run: npm run build
      
      - name: copy file via ssh password
        uses: appleboy/scp-action@v0.1.4
        with:
          host: ${{ secrets.SSH_HOST }}
          key: ${{ secrets.SSH_KEY }}
          username: ${{ secrets.SSH_USERNAME }}
          source: build/*
          target: ../var/www/staging.premed.pk/html/
          strip_components: 1
